MAIN METHOD PARAMETERS

Valid Main Method Signatures:

// Standard Main method
static void Main() { }

// Main method with arguments (command-line arguments)
static void Main(string[] args) { }

// Main method with arguments and return type (int) - used to return an exit code
static int Main(string[] args) { return 0; }
