// Import necessary namespaces

// Provides MVC features like Controller base class, IActionResult, etc.
using Microsoft.AspNetCore.Mvc;

// Imports the Student model class from the Models folder
using RoutingInASPDotNetCoreMVC.Models;

namespace RoutingInASPDotNetCoreMVC.Controllers // Defines the Controllers namespace
{
    // StudentController is a controller that handles HTTP requests for "Student" data
    // Inherits from Controller class (base class for all MVC controllers in ASP.NET Core)
    public class StudentController : Controller
    {
        // -------------------------------------------------------------------------
        // Static list of students acting as an in-memory "data source"
        // In real-world applications, data would be fetched from a database
        // -------------------------------------------------------------------------
        static List<Student> students = new List<Student>()
        {
            new Student() { Id = 1, Name = "Pranaya" },
            new Student() { Id = 2, Name = "Priyanka" },
            new Student() { Id = 3, Name = "Anurag" },
            new Student() { Id = 4, Name = "Sambit" }
        };

        // -------------------------------------------------------------------------
        // Method: GetAllStudents
        // Route: /Student/All
        // Returns a list of all students

        [Route("Student/All")] // Attribute routing: maps /Student/All to this method
        public List<Student> GetAllStudents()
        {
            // Return the entire student list
            return students;
        }

        // -------------------------------------------------------------------------
        // Method: GetStudentByID
        // Route: /Student/{studentID}/Details
        // Fetches a student by ID from the list and returns it
        // -------------------------------------------------------------------------

        [Route("Student/{studentID}/Details")] // {studentID} is a route parameter
        public Student GetStudentByID(int studentID) // ASP.NET Core binds the value from the URL to this method parameter
        {
            // FirstOrDefault returns the first matching student or null
            Student? studentDetails = students.FirstOrDefault(s => s.Id == studentID);

            // If student is not found, return a new empty Student object (to avoid null)
            return studentDetails ?? new Student();
        }

                // Method: GetStudentCourses
        // Route: /Student/{studentID}/Courses
        // Returns a list of course names based on the student ID

        [Route("Student/{studentID}/Courses")] // Route template with variable studentID
        public List<string> GetStudentCourses(int studentID)
        {
            // Initialize course list
            List<string> CourseList = new List<string>();

            // Hardcoded course data for different students
            if (studentID == 1)
                CourseList = new List<string>() { "ASP.NET Core", "C#.NET", "SQL Server" };
            else if (studentID == 2)
                CourseList = new List<string>() { "ASP.NET Core MVC", "C#.NET", "ADO.NET Core" };
            else if (studentID == 3)
                CourseList = new List<string>() { "ASP.NET Core WEB API", "C#.NET", "Entity Framework Core" };
            else
                CourseList = new List<string>() { "Bootstrap", "jQuery", "AngularJs" };

            // Return course list based on studentID
            return CourseList;
        }
    }
}

ğŸ—‚Program.cs
ğŸ‘‡ Code:
    public class Program
    {
        public static void Main(string[] args)
        {
            // Create the WebApplicationBuilder instance
            var builder = WebApplication.CreateBuilder(args);

            // Register MVC services into the dependency injection container
            // This adds controllers, views, model binding, data annotations, etc.
            builder.Services.AddControllersWithViews();

            // Build the application
            var app = builder.Build();

            // Configure the HTTP request pipeline
            if (!app.Environment.IsDevelopment())
            {
                // If the app is NOT in development, use a custom error page
                app.UseExceptionHandler("/Home/Error");
                app.UseHsts(); // Adds HTTP Strict Transport Security header
            }

            // Redirect HTTP to HTTPS
            app.UseHttpsRedirection();

            // Serve static files from wwwroot folder
            app.UseStaticFiles();

            // Enable routing capabilities
            app.UseRouting();

            // Enable authorization middleware (not needed here but standard)
            app.UseAuthorization();

            // Use attribute routing with controllers
            // This tells ASP.NET Core to look for [Route] attributes in controllers
            app.MapControllers(); //We are not using app.MapControllersRoute() middlewere becoz the routes are defined in the controllers using the concept of attribute routing.

            // Run the app
            app.Run();
        }
}
